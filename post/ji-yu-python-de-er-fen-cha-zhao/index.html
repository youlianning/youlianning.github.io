基于python的二分查找

2022-02-22
list1 = [1, 2, 3, 4, 5, 6]


def binary_search(list_v, item):
    low = 0
    high = len(list_v) - 1
    while low <= high:
        mid = int((low + high) / 2)
        guess = list_v[mid]
        if guess == item:
            return mid
        if guess > item:
            high = mid - 1
        else:
            low = mid + 1
    return None

print(binary_search(list1,3))
