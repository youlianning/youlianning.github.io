基于python的求某数在其若干个质因数相乘下的商

2022-02-22 python 
from itertools import combinations,permutations

list1 = [2, 2, 2, 3, 3, 3, 5]
list2=[]
for n in range(7):
    c = list(set(combinations(list1,n+1)))
    for key in c:
        mul = 1
        for i in key:
            mul = mul * i
        list2.append(mul)
list2.sort()
for item in list2:
    print("1080 / ",item,'=',1080/item)
